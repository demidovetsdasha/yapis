for i in 1..5:
    insert to numbers:
        n = i

write numbers

switch x:
    case is true:
        write "true"
    default:
        write "false"

select n from numbers as selected
for x in selected:
    if x % 2 == 0:
        write "even:"
        write x
    else:
        write "odd:"
        write x
